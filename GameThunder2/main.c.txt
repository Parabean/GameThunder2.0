#include "game.h"

int main()
{
	int quit = 0;
	while (!quit)
	{
		Menu();
		int select = 0;
		scanf("%d", &select);
		switch (select)
		{
		case 1:
			Play();
			break;
		case 2:
			quit = 1;
			break;
		default:
			printf("您的输入有误，请重新输入！\n");
			break;
		}
	}
	printf("ByeBye!");
	system("pause");
	return 0;
}
